FundamentosIntroducción al lenguaje C

Estructura de un programa

Todo programa en C tiene una función de entrada obligatoria: main(). El preprocesador gestiona las directivas #include y #define antes de la compilación.

C — estructura mínima 
#include <stdio.h>   /* librería estándar de E/S */
#include <stdlib.h>  /* malloc, free, exit…     */

/* Declaración de función (prototipo) */
void saludar(void);

int main(int argc, char *argv[]) {
    saludar();
    return 0;          /* 0 indica éxito al SO */
}

void saludar(void) {
    printf("Hola, mundo!\n");
}

Tipos de ficheros

.c — Fuente

Implementación: contiene las definiciones de funciones y variables globales.

.h — Cabecera

Interfaz pública: prototipos, macros, tipos y declaraciones extern.

.o — Objeto

Código máquina de un .c compilado, sin enlazar todavía.

a.out / bin — Ejecutable

Resultado del enlazado de uno o varios ficheros objeto.

Compilación básica con GCC

bash — comandos de compilación 
# Compilar directo a ejecutable
gcc main.c -o programa

# Activar advertencias (recomendado)
gcc -Wall -Wextra -std=c99 main.c -o programa

# Compilación en 2 pasos: .c → .o → ejecutable
gcc -c main.c -o main.o       # genera objeto
gcc main.o -o programa         # enlaza

# Multifuente
gcc -c main.c utils.c
gcc main.o utils.o -o programa
💡 Buena práctica Usa siempre -Wall -Wextra. GCC te avisará de variables no usadas, comparaciones peligrosas y muchos errores comunes antes de que sean bugs en producción.

Extern

extern declara que una variable existe y está definida en otro fichero fuente. Es la forma de compartir variables globales entre varios .c.

C — extern 
/* config.c — DEFINICIÓN (asigna espacio) */
int modo_debug = 0;

/* utils.c — DECLARACIÓN (solo avisa al compilador) */
extern int modo_debug;   /* sin reservar memoria */

void log(const char *msg) {
    if (modo_debug) fprintf(stderr, "[DBG] %s\n", msg);
}

C
int dia = 3;

switch (dia) {
    case 1:  printf("Lunes\n");     break;
    case 2:  printf("Martes\n");    break;
    case 3:  printf("Miércoles\n"); break;
    case 6:
    case 7:  printf("Fin de semana\n"); break;

for / while / do-while

C — bucles
/* for: cuando se conoce el número de iteraciones */
for (int i = 0; i < 5; i++) {
    printf("%d ", i);   /* 0 1 2 3 4 */
}

/* while: condición evaluada ANTES de cada iteración */
int n = 10;
while (n > 0) {
    n -= 3;
}  /* n termina en -2 */

/* do-while: se ejecuta AL MENOS una vez */
int opcion;
do {
    printf("Elige 1–3: ");
    scanf("%d", &opcion);
} while (opcion < 1 || opcion > 3);

/* break y continue */
for (int i = 0; i < 10; i++) {
    if (i == 5) break;     /* sale del bucle al llegar a 5 */
    if (i % 2 == 0) continue; /* salta pares, sigue iterando */
    printf("%d ", i);   /* imprime: 1 3 */
}

MemoriaPunteros

Un puntero almacena la dirección de memoria de otra variable. Son la base de la gestión dinámica, los arrays y el paso por referencia en C.

C — operadores * y &
int  valor = 42;
int *ptr;          /* declara puntero a int */

ptr = &valor;       /* & → "dirección de" (0x7fff...) */

printf("%p\n", ptr);     /* dirección: 0x7fff... */
printf("%d\n", *ptr);   /* * → "valor en esa dir": 42 */

*ptr = 100;             /* modifica valor a través del puntero */
printf("%d\n", valor);  /* imprime: 100 */

/* Puntero nulo — inicializar siempre si no se usa */
int *nulo = NULL;       /* nunca desreferenciar NULL */

Aritmética de punteros

C
int arr[] = { 10, 20, 30, 40 };
int *p = arr;       /* apunta al primer elemento */

printf("%d\n", *p);         /* 10 */
printf("%d\n", *(p+1));     /* 20 — avanza sizeof(int) bytes */
printf("%d\n", *(p+2));     /* 30 */

p++;                       /* p ahora apunta a arr[1] */

/* Diferencia entre punteros (número de elementos) */
int *fin = arr + 4;
printf("%td\n", fin - arr);  /* 4 (tipo ptrdiff_t) */
⚠ Atención Sumar 1 a un int* avanza 4 bytes (tamaño de int), no 1 byte. La aritmética de punteros está indexada por el tamaño del tipo apuntado.

Paso por valor vs. referencia

C — paso de parámetros
/* Por VALOR: se copia el argumento, el original NO cambia */
void doble_valor(int x) {
    x *= 2;   /* solo modifica la copia local */
}

/* Por REFERENCIA (pasando puntero): modifica el original */
void doble_ref(int *x) {
    *x *= 2;  /* modifica el valor en la dirección apuntada */
}

int main() {
    int n = 5;
    doble_valor(n);        /* n sigue siendo 5 */
    doble_ref(&n);         /* n ahora es 10 */
    return 0;
}

HeapMemoria dinámica

La memoria dinámica se gestiona manualmente en el heap. A diferencia de las variables locales (stack), persiste hasta que se libera con free().

FunciónDescripción
malloc(size)Reserva size bytes. Contenido indeterminado.
calloc(n, size)Reserva n × size bytes, inicializados a 0.
realloc(ptr, size)Redimensiona un bloque existente.
free(ptr)Libera la memoria reservada. Esencial para evitar leaks.
sizeof(tipo)Tamaño en bytes de un tipo en tiempo de compilación.
strdup(s)Copia una cadena al heap (malloc + strcpy). Liberar con free.
C — malloc / calloc / realloc / free
#include <stdlib.h>
#include <string.h>

/* malloc: reservar array dinámico */
int n = 10;
int *arr = (int *) malloc(n * sizeof(int));
if (!arr) { perror("malloc"); return 1; }

/* calloc: ya inicializado a 0 */
int *zeros = (int *) calloc(n, sizeof(int));

/* realloc: ampliar a 20 elementos */
arr = (int *) realloc(arr, 20 * sizeof(int));
if (!arr) { perror("realloc"); return 1; }

/* strdup */
char *copia = strdup("texto original");

/* SIEMPRE liberar */
free(arr);
free(zeros);
free(copia);
arr = NULL;    /* evitar puntero colgante */

Operaciones sobre bloques de memoria

C — memcpy / memset / memcmp
#include <string.h>

int src[5] = { 1, 2, 3, 4, 5 };
int dst[5];

/* memcpy: copia n bytes de src a dst (no solapados) */
memcpy(dst, src, 5 * sizeof(int));

/* memset: rellena n bytes con un valor (útil para poner a 0) */
memset(dst, 0, 5 * sizeof(int));   /* dst = {0,0,0,0,0} */

/* memcmp: compara n bytes; 0 si iguales */
int igual = (memcmp(src, dst, 5 * sizeof(int)) == 0);
/* igual = 0 (son distintos tras el memset) */
⚠ Memory leaks Cada malloc/calloc/strdup debe tener su free correspondiente. Herramientas como Valgrind o AddressSanitizer (-fsanitize=address) detectan fugas de memoria automáticamente.
💡 Patrón seguro Después de un free(ptr), asigna siempre ptr = NULL para evitar que sea un puntero colgante (dangling pointer) que podría desreferenciarse accidentalmente.
